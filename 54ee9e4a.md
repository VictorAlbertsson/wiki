# Programming Language

```
%namespace("boat")

boat_type = enum (
    KETCH
    YAWL
    SLOOP
    SQRIG
    MOTOR
)

boat = struct (
    wll = int // water line length
    t = boat_type // type
    sa = lng // sail area
)

motorboat(x=int,y=lng) = boat(wll = x, t = boat_type.MOTOR, sa = y)

winners = array (
    boat(40, boat_type.YAWL, 6_000_000L)
    motorboat(28,0)
    boat(0)
)
```
